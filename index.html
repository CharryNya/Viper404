<script>

// Función simple para cambiar estado de botón me gusta / no me gusta

function toggleReaction(el) {

const isActive = el.classList.contains('active');

// Desactivar todos los hermanos

el.parentNode.querySelectorAll('.reaction-btn').forEach(btn => btn.classList.remove('active'));

if (!isActive) el.classList.add('active');

}
